00026 /* Select source for FLLREF  e.g. SELECT_FLLREF(SELREF__XT1CLK) */
00027 #define SELECT_FLLREF(source) st(UCSCTL3 = (UCSCTL3 & ~(SELREF_7)) | (source);) 
00028 /* Select source for ACLK    e.g. SELECT_ACLK(SELA__XT1CLK) */
00029 #define SELECT_ACLK(source)   st(UCSCTL4 = (UCSCTL4 & ~(SELA_7))   | (source);) 
00030 /* Select source for MCLK    e.g. SELECT_MCLK(SELM__XT2CLK) */
00031 #define SELECT_MCLK(source)   st(UCSCTL4 = (UCSCTL4 & ~(SELM_7))   | (source);) 
00032 /* Select source for SMCLK   e.g. SELECT_SMCLK(SELS__XT2CLK) */
00033 #define SELECT_SMCLK(source)  st(UCSCTL4 = (UCSCTL4 & ~(SELS_7))   | (source);) 
00034 /* Select source for MCLK and SMCLK e.g. SELECT_MCLK_SMCLK(SELM__DCOCLK + SELS__DCOCLK) */
00035 #define SELECT_MCLK_SMCLK(sources) st(UCSCTL4 = (UCSCTL4 & ~(SELM_7 + SELS_7)) | (sources);)
